每日一解

反转一个单链表。
示例:
输入: 1->2->3->4->5->NULL
输出: 5->4->3->2->1->NULL

class Solution:
          def reverseList(self, head: ListNode) -> ListNode:
              """
              target:游标点，记录当下读取位置
              left:链表最左边点
              right:可用于交换的游标点
              """
              if head:
                  target = head
                  left = head
                  right = head
      
                  while True:
                      if target.next:
                          right = target
                          target  = target.next
                          right.next = left
                          left = right
                      else:
                          right = target
                          right.next = left
                          head.next = None
                          return target
              else:
                  return head
